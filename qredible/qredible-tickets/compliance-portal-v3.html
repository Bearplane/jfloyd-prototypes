Engineer · QP-3585 · Dual-flag focus feature schema + invite resolution
// SCHEMA — extends Q-Config Feature Limits architecture. Two new columns.

// 1. Feature gets a new admin-toggle flag:
ALTER TABLE pl_feature
  ADD COLUMN can_be_focus BOOLEAN NOT NULL DEFAULT FALSE;

// 2. Plan picks one of its bound features as the focus.
//    The bound-features relation already exists in Q-Config (plan ↔ features many-to-many).
//    The focusFeatureId is just an FK pointing into that set, constrained.
ALTER TABLE pl_plan
  ADD COLUMN focus_feature_id UUID REFERENCES pl_feature(id);

// Constraint: focus_feature_id must
//   (a) be in the plan's bound features (enforce in app or via composite FK)
//   (b) reference a feature where can_be_focus = TRUE (enforce in app + Plan editor UI)

// TypeScript shapes:
type Feature = {
  id: string;
  key: string;                    // "COMPLIANCE_CASES", "MYCOA", etc.
  name: string;
  canBeFocus: boolean;            // NEW — admin toggle on Feature Builder Details tab
  // ... existing limits, triggers, etc. from Q-Config Feature Limits work
};

type Plan = {
  id: string;
  name: string;
  boundFeatureIds: string[];      // existing — many-to-many with Feature
  focusFeatureId: string | null;  // NEW — FK to one of boundFeatureIds, must be canBeFocus=true
  // ... existing price, tier, etc.
};
